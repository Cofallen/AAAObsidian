```dataviewjs
// =======================
// 数学知识树 v1.0
// Author: ChatGPT
// =======================

const ROOT_TAG = "#数学";
const FOLDER = "MyLife/Daily";

// ------------------------
// 标签树
// ------------------------

const tree = {};

function insert(parts, obj) {

    let node = tree;

    for (const part of parts) {

        if (!node[part]) {
            node[part] = {
                children: {},
                items: []
            };
        }

        node = node[part].children;
    }

    // 找到最终节点
    node = tree;

    for (let i = 0; i < parts.length; i++) {

        node = node[parts[i]];

        if (i < parts.length - 1)
            node = node.children;
    }

    node.items.push(obj);

}

// ------------------------
// 扫描所有 Daily
// ------------------------

const pages = dv.pages(`"${FOLDER}"`)
    .sort(p => p.file.day, "desc");

for (const page of pages) {

    const file = app.vault.getAbstractFileByPath(page.file.path);

    if (!file) continue;

    const text = await app.vault.cachedRead(file);

    const lines = text.split(/\r?\n/);

    for (let line of lines) {

        line = line.trim();

        if (!line) continue;

        // 提取所有标签
        const tags = line.match(/#[^\s#]+/g);

        if (!tags) continue;

        for (const tag of tags) {

            if (!tag.startsWith(ROOT_TAG))
                continue;

            const parts = tag.substring(1).split("/");

            insert(parts, {

                file: page.file.link,

                day: page.file.day,

                text: line

            });

        }

    }

}

// ------------------------
// 渲染
// ------------------------

function count(node){

    let total = node.items.length;

    for(const child of Object.values(node.children))
        total += count(child);

    return total;

}

function render(node, depth){

    const keys = Object.keys(node).sort();

    for(const key of keys){

        const current=node[key];

        const total=count(current);

        dv.header(depth, `📂 ${key} (${total})`);

        if(current.items.length){

            dv.table(
                ["日期","知识"],

                current.items.map(x=>[
                    x.file,
                    x.text
                ])

            );

        }

        render(current.children,depth+1);

    }

}

render(tree,2);
```
